cached_stop_words = stopwords.words("english")

word_list = []
for tweet in tweet_list:
    for word in tweet.split():
        word = re.sub('^u\'','',word)
        word = re.sub('\'','',word)
        word = word.lower()
        if len(word) > 3:
            if str(word) not in cached_stop_words:
                word_list.append(word)
